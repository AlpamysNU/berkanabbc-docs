    TelegramCode — временные коды авторизации¶

    class TelegramCode(models.Model):
    telegram_username = models.CharField(max_length=64)    # @username
    telegram_id = models.BigIntegerField()                 # числовой ID Telegram
    code = models.CharField(max_length=4)                  # "1234"
    expires_at = models.DateTimeField()                    # now() + 10 мин
    is_used = models.BooleanField(default=False)
    created_at = models.DateTimeField(auto_now_add=True)

    class Meta:
        indexes = [Index(fields=['telegram_username', 'code'])]

    Очистка: записи с is_used=True и expires_at < now() удаляются периодически.

    Child — профиль ребёнка¶

    class Child(models.Model):
    GENDERS = [('m', 'Мальчик'), ('f', 'Девочка'), ('o', 'Другое')]

    parent = models.ForeignKey(UserProfile, on_delete=CASCADE, related_name='children')
    full_name = models.CharField(max_length=200)
    last_name = models.CharField(max_length=100, blank=True)
    first_name = models.CharField(max_length=100, blank=True)
    patronymic = models.CharField(max_length=100, blank=True)
    date_of_birth = models.DateField()
    gender = models.CharField(max_length=1, choices=GENDERS)
    has_attached_doctor = models.BooleanField(default=False)
    doctor_full_name = models.CharField(max_length=200, blank=True)
    doctor_contacts = models.CharField(max_length=200, blank=True)
    doctor_workplace = models.CharField(max_length=200, blank=True)
    doctor_specialization = models.CharField(max_length=100, blank=True)
    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)

    def save(self, *args, **kwargs):
        # full_name собирается из ФИО для обратной совместимости старого UI/API.
        parts = [self.last_name, self.first_name, self.patronymic]
        composed_name = ' '.join(part for part in parts if part).strip()
        if composed_name:
            self.full_name = composed_name
        super().save(*args, **kwargs)

    @property
    def age_months(self):  # было age_in_months — исправлено 2026-05-03
        today = date.today()
        dob = self.date_of_birth
        return (today.year - dob.year) * 12 + (today.month - dob.month)

    Report — отчёт о скрининге¶

    class Report(models.Model):

    compared_to_session = models.ForeignKey(
        ScreeningSession, null=True, blank=True, 
        on_delete=SET_NULL, related_name='+'
    )
    # {"motor": +5, "cognitive": -3, ...}
    delta_scores = models.JSONField(null=True, blank=True)
    generated_at = models.DateTimeField(auto_now_add=True)
